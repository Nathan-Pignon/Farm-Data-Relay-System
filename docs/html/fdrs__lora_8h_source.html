163// crc16_update used by both LoRa and filesystem
164
165// CRC16 from https://github.com/4-20ma/ModbusMaster/blob/3a05ff87677a9bdd8e027d6906dc05ca15ca8ade/src/util/crc16.h#L71

177static uint16_t crc16_update(uint16_t crc, uint8_t a)
178{
179 int i;
180
181 crc ^= a;
182 for (i = 0; i < 8; ++i)
183 {
184 if (crc & 1)
185 crc = (crc >> 1) ^ 0xA001;
186 else
187 crc = (crc >> 1);
188 }
189
190 return crc;
191}

205// Do not call this function directly, instead call transmitLoRaAsync(...)
206// Transmits Lora data by calling RadioLib library function
207// Returns void becuase function is Async and does not receive any data
208void transmitLoRa(uint16_t *destMac, DataReading *packet, uint8_t len)
209{
210 uint8_t pkt[6 + (len * sizeof(DataReading))];
211 uint16_t calcCRC = 0x0000;
212 loraTxState = stInProcess;
213 // Building packet -- address portion - first 4 bytes
214 pkt[0] = (*destMac >> 8); // high byte of destination MAC
215 pkt[1] = (*destMac & 0x00FF); // low byte of destination MAC
216 pkt[2] = selfAddress[4]; // high byte of source MAC (ourselves)
217 pkt[3] = selfAddress[5]; // low byte of source MAC
218 // Building packet -- data portion - 7 bytes
219 memcpy(&pkt[4], packet, len * sizeof(DataReading)); // copy data portion of packet
220 // Calculate CRC of address and data portion of the packet
221 // Last 2 bytes are CRC so do not include them in the calculation itself
222 for (int i = 0; i < (sizeof(pkt) - 2); i++)
223 {
224 // printf("CRC: %02X : %d\n",calcCRC, i);
225 calcCRC = crc16_update(calcCRC, pkt[i]);
226 }
227 if(!ack) {
228 calcCRC = crc16_update(calcCRC, 0xA1); // Recalculate CRC for No ACK
229 }
230 pkt[len * sizeof(DataReading) + 4] = (calcCRC >> 8); // Append calculated CRC to the last 2 bytes of the packet
231 pkt[len * sizeof(DataReading) + 5] = (calcCRC & 0x00FF);
232 DBG("Sending LoRa DR");
233 DBG1("Transmitting LoRa message of size " + String(sizeof(pkt)) + " bytes with CRC 0x" + String(calcCRC, HEX) + " to destination 0x" + String(*destMac, HEX));
234 // printLoraPacket(pkt,sizeof(pkt));
235 tx_start_time = millis();
236 int state = radio.startTransmit(pkt, sizeof(pkt));
237 transmitFlag = true;
238 if (state != RADIOLIB_ERR_NONE)
239 {
240 DBG("Xmit failed, code " + String(state));
241 while (true)
242 ;
243 }
244 txCountDR++;
245 return;
246}

340bool transmitLoRaAsync(uint16_t *destAddr, SystemPacket *sp, uint8_t len)
341{
342 for(int i=0; i < len; i++)
343 {
344 //check for full buffer
345 if(ISBUFFFULL(spBuff)) {
346 DBG("Lora SP Buffer Overflow!");
347 BUFFINCSTART(spBuff);
348 }
349
350 //add packet to buffer
351 *(spBuff.sp + spBuff.endIdx) = *(sp + i);
352 *(spBuff.address + spBuff.endIdx) = *destAddr;
353 BUFFINCEND(spBuff);
354 }
355 DBG2("SP added to LoRa buffer. start: " + String(spBuff.startIdx) + " end: " + String(spBuff.endIdx));
356 return true;
357}

461// ****DO NOT CALL receiveLoRa() directly! ***** Call handleLoRa() instead!
462// receiveLoRa for Sensors
463// USED to get ACKs (SystemPacket type) from LoRa gateway at this point. May be used in the future to get other data
464// Return type is crcResult struct - CRC_OK, CRC_BAD, CRC_NULL. CRC_NULL used for non-ack data
465
466crcResult receiveLoRa()
467{
468 int packetSize = radio.getPacketLength();
469 if ((((packetSize - 6) % sizeof(DataReading) == 0) || ((packetSize - 6) % sizeof(SystemPacket) == 0)) && packetSize > 0)
470 { // packet size should be 6 bytes plus multiple of size of DataReading
471 uint8_t packet[packetSize];
472 uint16_t packetCRC = 0x0000; // CRC Extracted from received LoRa packet
473 uint16_t calcCRC = 0x0000; // CRC calculated from received LoRa packet
474 uint16_t sourceMAC = 0x0000;
475 uint16_t destMAC = 0x0000;
476
477 radio.readData((uint8_t *)&packet, packetSize);
478
479 destMAC = (packet[0] << 8) | packet[1];
480 sourceMAC = (packet[2] << 8) | packet[3];
481 packetCRC = ((packet[packetSize - 2] << 8) | packet[packetSize - 1]);
482 // Print all packets
483 // DBG2("Source Address: 0x" + String(packet[2], HEX) + String(packet[3], HEX) + " Destination Address: 0x" + String(packet[0], HEX) + String(packet[1], HEX));
484
485#ifdef FDRS_GATEWAY
486 // for gateway - only listen to our own address
487 if (destMAC == (selfAddress[4] << 8 | selfAddress[5]))
488#elif defined(FDRS_NODE)
489 // for Node - need to listen to broadcast
490 if (destMAC == (selfAddress[4] << 8 | selfAddress[5]) || (destMAC == 0xFFFF))
491#endif
492 { // Check if addressed to this device or broadcast
493 // printLoraPacket(packet,sizeof(packet));
494 DBG1("Incoming LoRa. Size: " + String(packetSize) + " Bytes, RSSI: " + String(radio.getRSSI()) + "dBm, SNR: " + String(radio.getSNR()) + "dB, PacketCRC: 0x" + String(packetCRC, HEX));
495 // Evaluate CRC
496 for (int i = 0; i < (packetSize - 2); i++)
497 { // Last 2 bytes of packet are the CRC so do not include them in calculation
498 // printf("CRC: %02X : %d\n",calcCRC, i);
499 calcCRC = crc16_update(calcCRC, packet[i]);
500 }
501 if ((packetSize - 6) % sizeof(DataReading) == 0)
502 { // DataReading type packet
503 rxCountDR++;
504 if (calcCRC == packetCRC)
505 { // We've received a DR and sending an ACK
506 SystemPacket ACK = {.cmd = cmd_ack, .param = CRC_OK};
507 DBG1("CRC Match, sending ACK packet to node 0x" + String(sourceMAC, HEX) + "(hex)");
508 transmitLoRaAsync(&sourceMAC, &ACK, 1); // Send ACK back to source
509 }
510 else if (packetCRC == crc16_update(calcCRC, 0xA1))
511 { // Sender does not want ACK and CRC is valid
512 DBG1("Node address 0x" + String(sourceMAC, 16) + " does not want ACK");
513 }
514 else
515 { // We've received a DR and CRC is bad
516 SystemPacket NAK = {.cmd = cmd_ack, .param = CRC_BAD};
517 // Send NAK packet
518 DBG1("CRC Mismatch! Packet CRC is 0x" + String(packetCRC, HEX) + ", Calculated CRC is 0x" + String(calcCRC, HEX) + " Sending NAK packet to node 0x" + String(sourceMAC, HEX) + "(hex)");
519 transmitLoRaAsync(&sourceMAC, &NAK, 1); // CRC did not match so send NAK to source
520 newData = event_clear; // do not process data as data may be corrupt
521 return CRC_BAD; // Exit function and do not update newData to send invalid data further on
522 }
523 rxCountCrcOk++;
524 memcpy(&theData, &packet[4], packetSize - 6); // Split off data portion of packet (N - 6 bytes (6 bytes for headers and CRC))
525 ln = (packetSize - 6) / sizeof(DataReading);
526 if (memcmp(&sourceMAC, &LoRa1, 2) == 0)
527 { // Check if it is from a registered sender
528 newData = event_lora1;
529 return CRC_OK;
530 }
531 else if (memcmp(&sourceMAC, &LoRa2, 2) == 0)
532 {
533 newData = event_lora2;
534 return CRC_OK;
535 }
536 else {
537 newData = event_lorag;
538 return CRC_OK;
539 }
540 }
541 else if ((packetSize - 6) == sizeof(SystemPacket))
542 {
543 unsigned int ln = (packetSize - 6) / sizeof(SystemPacket);
544 SystemPacket receiveData[ln];
545 rxCountSP++;
546 if ((packetCRC == calcCRC) || (packetCRC == crc16_update(calcCRC, 0xA1)))
547 {
548 memcpy(receiveData, &packet[4], packetSize - 6); // Split off data portion of packet (N bytes)
549 if (ln == 1 && receiveData[0].cmd == cmd_ack)
550 {
551 DBG1("ACK Received - CRC Match");
552 if(loraAckState == stInProcess) {
553 loraAckState = stCrcMatch;
554 }
555 }
556 else if (ln == 1 && receiveData[0].cmd == cmd_ping)
557 { // We have received a ping request or reply??
558 if (receiveData[0].param == ping_reply)
559 { // This is a reply to our ping request
560 loraPing.status = stCompleted;
561 DBG1("Ping reply via LoRa from address 0x" + String(sourceMAC, HEX));
562 }
563 else if (receiveData[0].param == ping_request)
564 {
565 DBG1("Ping request from 0x" + String(sourceMAC, HEX) + ", Replying.");
566 pingReplyLoRa(sourceMAC);
567 }
568 }

671// FDRS Sensor pings address and listens for a defined amount of time for a reply if no tx in process
672// otherwise queues up a ping in the SP Buffer.
673int pingRequestLoRa(uint16_t address, uint32_t timeout)
674{
675 int pingResult = -1;
676
677 // Check if a previous ping is already in process
678 if(loraPing.status == stReady) {
679 SystemPacket sys_packet = {.cmd = cmd_ping, .param = ping_request};
680
681 loraPing.timeout = timeout;
682 loraPing.address = address;
683 // Perform blocking ping if nothing else is in process
684 if(loraTxState == stReady) {
685 loraPing.status = stInProcess;
686 loraPing.start = millis();
687 DBG1("LoRa ping request sent to address: 0x" + String(address, HEX));
688 transmitLoRa(&address,&sys_packet,1);
689 while(loraPing.status == stInProcess && (millis() - loraPing.start < loraPing.timeout)) {
690 LoRaTxRxOperation();
691 }
692 if(loraPing.status == stCompleted) {
693 loraPing.response = millis() - loraPing.start;
694 pingResult = loraPing.response;
695 DBG("LoRa Ping Returned: " + String(loraPing.response) + "ms.");
696 if(loraPing.address == timeSource.tmAddress) {
697 netTimeOffset = loraPing.response/2/1000;
698 adjTimeforNetDelay(netTimeOffset);
699 }
700 }
701 else {
702 DBG("No LoRa ping returned within " + String(loraPing.timeout) + "ms.");
703 }
704 loraPing.status = stReady;
705 loraPing.start = 0;
706 loraPing.timeout = 0;
707 loraPing.address = 0;
708 loraPing.response = UINT32_MAX;
709 }
710 // Something else is in process, Most likely LoRa ACK, so queue up the ping
711 else {
712 if(transmitLoRaAsync(&address, &sys_packet, 1))
713 {
714 DBG1("LoRa ping request queued to address: 0x" + String(address, HEX));
715 return true;
716 }
717 else {
718 DBG1("Error sending LoRa ping.");
719 }
720 }
721 }
722 return pingResult;
723}

751void handleLoRa()
752{
753 static uint8_t len = 0;
754 static DataReading *data;
755 static uint16_t address;
756 static unsigned long lastTxtime = 0;
757 static unsigned long statsTime = 0;
758
759 LoRaTxRxOperation();
760
761 // check for result of any ongoing async ping operations
762 if(loraPing.status == stCompleted) {
763 loraPing.response = millis() - loraPing.start;
764 DBG1("LoRa Ping Returned: " + String(loraPing.response) + "ms.");
765 if(loraPing.address == timeSource.tmAddress) {
766 netTimeOffset = loraPing.response/2/1000;
767 adjTimeforNetDelay(netTimeOffset);
768 }
769 loraPing.status = stReady;
770 loraPing.start = 0;
771 loraPing.timeout = 0;
772 loraPing.address = 0;
773 loraPing.response = UINT32_MAX;
774 }
775 if(loraPing.status == stInProcess && (TDIFF(loraPing.start,loraPing.timeout))) {
776 DBG1("No LoRa ping returned within " + String(loraPing.timeout) + "ms.");
777 loraPing.status = stReady;
778 loraPing.start = 0;
779 loraPing.timeout = 0;
780 loraPing.address = 0;
781 loraPing.response = UINT32_MAX;
782 }
783
784 static int retries = FDRS_LORA_RETRIES;
785
786 // Process any DR ACKs in progress
787 if(loraTxState == stReady && loraAckState != stReady) {
788 if (loraAckState == stCrcMatch)
789 {
790 DBG1("LoRa ACK Received! CRC OK");
791 free(data);
792 data = NULL;
793 retries = FDRS_LORA_RETRIES;
794 len = 0;
795 loraAckState = stReady;
796 }
797 else if(retries < 0) {
798 DBG1("Retries Exhausted.");
799 retries = FDRS_LORA_RETRIES;
800 if(ISBUFFFULL(drBuff)) {
801 len = 0;
802 free(data);
803 data = NULL;
804 }
805 // do we transmit ourselves to death or just drop the data?
806 // here we drop the data so we don't keep transmitting to death
807 len = 0;
808 free(data);
809 data = NULL;
810 loraAckState = stReady;
811 }
812 else if (loraAckState == stCrcMismatch)
813 {
814 DBG1("LoRa ACK Received! CRC BAD");
815 // Resend original packet again if retries are available
816 loraAckState = stReady;
817 }
818 else if (TDIFF(loraAckTimeout,FDRS_ACK_TIMEOUT)) {
819 DBG1("LoRa Timeout waiting for ACK!");
820 // resend original packet again if retries are available
821 loraAckState = stReady;
822 }
823 if(loraTxState == stCompleted) {
824 loraTxState = stReady;
825 }
826 return;
827 }
828
829 // Start Transmit data from the SystemPacket queue
830 if(!ISBUFFEMPTY(spBuff) && (loraTxState == stReady)) {
831 DBG2("SP Index: start: " + String(spBuff.startIdx) + " end: " + String(spBuff.endIdx) + " Address: 0x" + String(*(spBuff.address + spBuff.startIdx),HEX) + " Cmd: " + String(spBuff.sp->cmd));
832 // Lora ping request stuff here
833 if((spBuff.sp + spBuff.startIdx)->cmd == cmd_ping && (spBuff.sp + spBuff.startIdx)->param == ping_request) {
834 loraPing.status = stInProcess;
835 loraPing.start = millis();
836 DBG1("LoRa ping request sent to address: 0x" + String(*(spBuff.address + spBuff.startIdx), HEX));
837 }
838 transmitLoRa((spBuff.address + spBuff.startIdx), (spBuff.sp + spBuff.startIdx), 1);
839 BUFFINCSTART(spBuff);
840 }
841
842
843 // It's polite to Listen more than you talk
844 if(TDIFF(lastTxtime,(TXDELAYMS + random(0,50)))) {
845 // Start Transmit data from the DataReading queue
846 if((!ISBUFFEMPTY(drBuff) || (data != NULL)) && loraTxState == stReady && loraAckState == stReady)
847 {
848 // for(int i=drBuff.startIdx; i!=drBuff.endIdx; i = (i + 1) % drBuff.size) {
849 // printf("id: %d, type: %d data: %f address: %X\n",(drBuff.dr + i)->id, (drBuff.dr + i)->t, (drBuff.dr + i)->d, *(spBuff.address + spBuff.startIdx));
850 // }
851
852 // data memory is not freed when retries are exhausted and DataReading queue is not full
853 if(data == NULL) {
854 // Get number of DRs going to same destination address
855 len = transmitSameAddrLoRa();
856 // TransmitLoRa cannot handle a circular buffer so need data in one contiguous segment of memory
857 data = (DataReading *)malloc(len * sizeof(DataReading));
858 // Transfer data readings from the ring buffer to our local buffer
859 for(int i=0; i < len; i++) {
860 *(data + i) = *(drBuff.dr + ((drBuff.startIdx + i) % drBuff.size));
861 }
862 address = *(drBuff.address + drBuff.startIdx);
863 // now we have the data, we can release it from the ring buffer
864 drBuff.startIdx = (drBuff.startIdx + len) % drBuff.size;
865 }
866 DBG2("Length: " + String(len) + " Address: 0x" + String(address,HEX) + " Data:");
867 // for(int i=0; i< len; i++) {
868 // printf("id: %d, type: %d data: %f\n",(data + i)->id, (data + i)->t, (data + i)->d);
869 // }
870 if(!ack && data != NULL)
871 {
872 transmitLoRa(&address, data, len);
873 free(data);
874 data = NULL;
875 len = 0;
876 }
877 else if(data != NULL)
878 {
879 retries--;
880 loraAckState = stInProcess;
881 transmitLoRa(&address, data, len);
882 }
883 }
884
885 // Ping LoRa time master to estimate time delay in radio link
886 if(timeSource.tmNetIf == TMIF_LORA) {
887 static unsigned long lastTimeSourcePing = 0;
888
889 // ping the time source every 10 minutes
890 if(TDIFFMIN(lastTimeSourcePing,10) || lastTimeSourcePing == 0) {
891 pingRequestLoRa(timeSource.tmAddress,4000);
892 lastTimeSourcePing = millis();
893 }
894 }
895 lastTxtime = millis();
896 }
897 // Print LoRa statistics
898 if(TDIFFSEC(statsTime,65) && (rxCountDR + rxCountSP) > 0) {
899 statsTime = millis();
900 DBG1("LoRa Stats - Rx DR:" + String(rxCountDR) + " SP:" + String(rxCountSP) + " Tx DR:" + String(txCountDR) + " SP:" + String(txCountSP) + " CRC OK: " + String(rxCountCrcOk/(rxCountDR + rxCountSP) * 100) + "%");
901 }
902
903 // Change to ready at the end so only one transmit happens per function call
904 if(loraTxState == stCompleted) {
905 loraTxState = stReady;
906 }
907 return;
908}
