117void sendFDRS()

121 {
122 theData[i].id = fdrsData[i].id;
123 theData[i].t = fdrsData[i].t;
124 theData[i].d = fdrsData[i].d;

140void loadFDRS(float d, uint8_t t, uint16_t id)
141{
142 // Protéger contre le dépassement de tampon
143 if(data_count > 253) {
144 sendFDRS();
145 }
146 DBG("Id: " + String(id) + " - Type: " + String(t) + " - Data loaded: " + String(d));
147 DataReading dr;
148 dr.id = id;
149 dr.t = t;
150 dr.d = d;
151 fdrsData[data_count] = dr;
152 data_count++;
153}

218void handleCommands()
219{
220 switch (theCmd.cmd)
221 {
222 case cmd_ping:
223#ifdef USE_ESPNOW
224 pingback_espnow();
225#endif // USE_ESPNOW
226
227 break;
228
229 case cmd_add:
230#ifdef USE_ESPNOW
231 add_espnow_peer();
232#endif // USE_ESPNOW
233
234 break;
235
236 case cmd_time:
237 if(theCmd.param > MIN_TS) {
238#ifdef USE_ESPNOW
239 recvTimeEspNow(theCmd.param);
240#endif // USE_ESPNOW
241 }
242 else if(theCmd.param == 0) {
243#ifdef USE_ESPNOW
244 sendTimeESPNow(incMAC);
245#endif // USE_ESPNOW
246 }
247
248 break;
249 }
250 theCmd.cmd = cmd_clear;
251 theCmd.param = 0;
252}
