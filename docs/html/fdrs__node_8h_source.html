149void handleIncoming()
150{
151 if (newData != event_clear)
152 {
153
154 newData = event_clear;
155 for (int i = 0; i < ln; i++)
156 { // Cycle through array of incoming DataReadings for any we are subbed to
157 for (int j = 0; j < 255; j++)
158 { // Cycle through subscriptions for active entries
159 if (active_subs[j])
160 {
161
162 if (theData[i].id == subscription_list[j])
163 {
164
165 (*callback_ptr)(theData[i]);
166 }
167 }
168 }
169 }

173bool sendFDRS()
174{
175 if(data_count == 0) {
176 return false;
177 }
178 DBG("Sending FDRS Packet!");
179#ifdef USE_ESPNOW
180 esp_now_send(gatewayAddress, (uint8_t *)&fdrsData, data_count * sizeof(DataReading));
181 esp_now_ack_flag = CRC_NULL;
182 while (esp_now_ack_flag == CRC_NULL)
183 {
184 yield();
185 delay(0);
186 }
187 if (esp_now_ack_flag == CRC_OK)
188 {
189 data_count = 0;
190 return true;
191 }
192 else
193 {
194 data_count = 0;
195 return false;
196 }
197#endif
198#ifdef USE_LORA
199 transmitLoRaAsync(&gtwyAddress, fdrsData, data_count);
200 // DBG(" LoRa sent.");
201#ifdef LORA_ACK
202 if(crcReturned == CRC_OK) {
203 data_count = 0;
204 return true;
205 }
206#endif
207#ifndef LORA_ACK
208 if(crcReturned == CRC_OK || crcReturned == CRC_NULL) {
209 data_count = 0;
210 return true;
211}
212#endif
213 else {
214 data_count = 0;
215 return false;
216 }
217#endif
218}

288 is_controller = true;
289 callback_ptr = new_cb_ptr;

291 SystemPacket sys_packet = {.cmd = cmd_add, .param = 0};
292 DBG("ESP-NOW peer registration request submitted to 0x" + String(gatewayAddress[5],HEX));

338bool subscribeFDRS(uint16_t sub_id)
339{
340 for (int i = 0; i < 255; i++)
341 {
342 if ((subscription_list[i] == sub_id) && (active_subs[i]))
343 {
344 DBG("You're already subscribed to ID " + String(sub_id));
345 return true;
346 }
347 }
348 for (int i = 0; i < 255; i++)
349 {
350 if (!active_subs[i])
351 {
352 DBG("Subscribing to DataReading ID " + String(sub_id));
353 subscription_list[i] = sub_id;
354 active_subs[i] = true;
355 return true;
356 }
357 }
358 DBG("No subscription could be established!");
359 return false;
360}
361bool unsubscribeFDRS(uint16_t sub_id)
362{
363 for (int i = 0; i < 255; i++)
364 {
365 if ((subscription_list[i] == sub_id) && (active_subs[i]))
366 {
367 DBG("Removing subscription to ID " + String(sub_id));
368 active_subs[i] = false;
369 return true;
370 }
371 }
372 DBG("No subscription to remove");
373 return false;
374}
