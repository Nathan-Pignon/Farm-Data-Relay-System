192// send an NTP request to the time server at the given address
193void sendNTPpacket(const char * address) {
194 // set all bytes in the buffer to 0
195 memset(packetBuffer, 0, NTP_PACKET_SIZE);
196 // Initialize values needed to form NTP request
197 // (see URL above for details on the packets)
198 packetBuffer[0] = 0b11100011; // LI, Version, Mode
199 packetBuffer[1] = 0; // Stratum, or type of clock
200 packetBuffer[2] = 6; // Polling Interval
201 packetBuffer[3] = 0xEC; // Peer Clock Precision
202 // 8 bytes of zero for Root Delay & Root Dispersion
203 packetBuffer[12] = 49;
204 packetBuffer[13] = 0x4E;
205 packetBuffer[14] = 49;
206 packetBuffer[15] = 52;
207
208 // all NTP fields have been given values, now
209 // you can send a packet requesting a timestamp:
210 FDRSNtp.beginPacket(address, 123); // NTP requests are to port 123
211 FDRSNtp.write(packetBuffer, NTP_PACKET_SIZE);
212 FDRSNtp.endPacket();
213}

216//DBG("GetTime Function");
217 if(timeSource.tmSource <= TMS_NTP) {
218#ifdef USE_ETHERNET
219 if(eth_connected) {
220#elif defined(USE_WIFI)

228 for(i = 0; i < 800; i++) {

230 break;
231 delay(10);
232 }
233 if(i < 800) {
234 DBG2("Took " + String(i * 10) + "ms to get NTP response from " + String(timeServer) + ".");
235 NTPFetchFail = 0;
236 // We've received a packet, read the data from it
237 FDRSNtp.read(packetBuffer, NTP_PACKET_SIZE); // read the packet into the buffer
238
239 // the timestamp starts at byte 40 of the received packet and is four bytes,
240 // or two words, long. First, extract the two words:
241
242 unsigned long highWord = word(packetBuffer[40], packetBuffer[41]);
243 unsigned long lowWord = word(packetBuffer[42], packetBuffer[43]);
244 // combine the four bytes (two words) into a long integer
245 // this is NTP time (seconds since Jan 1 1900):
246 unsigned long secsSince1900 = highWord << 16 | lowWord;
247 //DBG("Seconds since Jan 1 1900 = " + String(secsSince1900));
248
249 // now convert NTP time into everyday time:
250 // Unix time starts on Jan 1 1970. In seconds, that's 2208988800:
251 const unsigned long seventyYears = 2208988800UL;
252 // subtract seventy years:
253 // now is epoch format - seconds since Jan 1 1970
254 now = secsSince1900 - seventyYears;
255 if(setTime(now)) {
256 timeSource.tmNetIf = TMIF_LOCAL;
257 timeSource.tmAddress = 0xFFFF;
258 timeSource.tmSource = TMS_NTP;
259 timeSource.tmLastTimeSet = millis();
260 DBG1("Time source is now local NTP");
261 } // UTC time
262 }
263 else {
264 DBG1("Timeout getting a NTP response.");
265 }
266 }
267 }
268 return;
269}
